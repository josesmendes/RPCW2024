## Quais as cidades de um determinado distrito
select ?distrito ?cidade where { 
	?cidade rdf:type :Cidades.
    ?cidade :pertenceA ?distrito
}order by ?distrito

## distribuição de cidades por distrito
select ?distrito (count(distinct ?s) as ?ncidades) where { 
	?s rdf:type :Cidades.
    ?s :pertenceA ?distrito.
}
group by ?distrito
order by desc(?ncidades)
